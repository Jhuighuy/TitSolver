  PRIVATE
    "-Wno-cast-qual"
    "-Wno-old-style-cast"
    "-Wno-redundant-decls"
)

  tit_py
  PUBLIC
    "PY_SSIZE_T_CLEAN"
    "NPY_NO_DEPRECATED_API=NPY_1_7_API_VERSION"
  PRIVATE
    "Py_LIMITED_API=0x030A0000" # Python 3.10.
)
