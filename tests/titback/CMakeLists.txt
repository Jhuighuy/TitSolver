  titback
  NAME "titback/exec"

  COMMAND titback -c "import pytit $<SEMICOLON> print(pytit)"
)

add_tit_test(
  titback
  NAME "titback/exec_failure"
  EXIT_CODE 1
  MATCH_STDERR "exec_failure_stderr.txt"
  COMMAND titback -c "import does_not_exist"
)
